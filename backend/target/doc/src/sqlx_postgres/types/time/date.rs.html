23impl Encode<'_, Postgres> for Date {
24    fn encode_by_ref(&self, buf: &mut PgArgumentBuffer) -> Result<IsNull, BoxDynError> {
25        // DATE is encoded as number of days since epoch (2000-01-01)
26        let days: i32 = (*self - PG_EPOCH).whole_days().try_into().map_err(|_| {
27            format!("value {self:?} would overflow binary encoding for Postgres DATE")
28        })?;
29        Encode::<Postgres>::encode(days, buf)
30    }
31
32    fn size_hint(&self) -> usize {
33        mem::size_of::<i32>()
34    }
35}
36
37impl<'r> Decode<'r, Postgres> for Date {
38    fn decode(value: PgValueRef<'r>) -> Result<Self, BoxDynError> {
39        Ok(match value.format() {
40            PgValueFormat::Binary => {
41                // DATE is encoded as the days since epoch
42                let days: i32 = Decode::<Postgres>::decode(value)?;
43                PG_EPOCH + Duration::days(days.into())
44            }
45
46            PgValueFormat::Text => Date::parse(
47                value.as_str()?,
48                &format_description!("[year]-[month]-[day]"),
49            )?,
50        })
51    }
52}
