8fn must_be_valid_iden(name: &str) -> bool {
9    // can only begin with [a-z_]
10    name.chars()
11        .take(1)
12        .all(|c| c == '_' || c.is_ascii_alphabetic())
13        && name.chars().all(|c| c == '_' || c.is_ascii_alphanumeric())
14}

16fn impl_iden_for_unit_struct(
17    ident: &proc_macro2::Ident,
18    new_iden: &str,
19) -> proc_macro2::TokenStream {
20    let prepare = if must_be_valid_iden(new_iden) {
21        quote! {
22            fn prepare(&self, s: &mut dyn ::std::fmt::Write, q: sea_orm::sea_query::Quote) {
23                write!(s, "{}", q.left()).unwrap();
24                self.unquoted(s);
25                write!(s, "{}", q.right()).unwrap();
26            }
27        }
28    } else {
29        quote! {}
30    };
31    quote! {
32        impl sea_orm::sea_query::Iden for #ident {
33            #prepare
34
35            fn unquoted(&self, s: &mut dyn ::std::fmt::Write) {
36                write!(s, #new_iden).unwrap();
37            }
38        }
39    }
40}

42fn impl_iden_for_enum(
43    ident: &proc_macro2::Ident,
44    variants: Punctuated<Variant, syn::token::Comma>,
45) -> proc_macro2::TokenStream {
46    let variants = variants.iter();
47    let mut all_valid = true;
48
49    let match_pair: Vec<TokenStream> = variants
50        .map(|v| {
51            let var_ident = &v.ident;
52            let var_name = if var_ident == "Table" {
53                ident
54            } else {
55                var_ident
56            };
57            let mut var_name = var_name.to_string().to_snake_case();
58            v.attrs
59                .iter()
60                .filter(|attr| attr.path().is_ident("sea_orm"))
61                .try_for_each(|attr| {
62                    attr.parse_nested_meta(|meta| {
63                        if meta.path.is_ident("iden") {
64                            let litstr: LitStr = meta.value()?.parse()?;
65                            var_name = litstr.value();
66                            all_valid &= must_be_valid_iden(var_name.as_str());
67                        } else {
68                            // Reads the value expression to advance the parse stream.
69                            // Some parameters do not have any value,
70                            // so ignoring an error occurred here.
71                            let _: Option<Expr> = meta.value().and_then(|v| v.parse()).ok();
72                        }
73                        Ok(())
74                    })
75                })
76                .expect("something something");
77            quote! { Self::#var_ident => write!(s, "{}", #var_name).unwrap() }
78        })
79        .collect();
80
81    let match_arms: TokenStream = quote! { #(#match_pair),* };
82
83    let prepare = if all_valid {
84        quote! {
85            fn prepare(&self, s: &mut dyn ::std::fmt::Write, q: sea_orm::sea_query::Quote) {
86                write!(s, "{}", q.left()).unwrap();
87                self.unquoted(s);
88                write!(s, "{}", q.right()).unwrap();
89            }
90        }
91    } else {
92        quote! {}
93    };
94
95    quote! {
96        impl sea_orm::sea_query::Iden for #ident {
97            #prepare
98
99            fn unquoted(&self, s: &mut dyn ::std::fmt::Write) {
100                match self {
101                    #match_arms
102                };
103            }
104        }
105    }
106}

108pub fn expand_derive_iden(input: DeriveInput) -> syn::Result<TokenStream> {
109    let DeriveInput { ident, data, .. } = input;
110
111    let mut new_iden: TokenStream = ident.to_string().to_snake_case().parse().unwrap();
112    input
113        .attrs
114        .iter()
115        .filter(|attr| attr.path().is_ident("sea_orm"))
116        .try_for_each(|attr| {
117            attr.parse_nested_meta(|meta| {
118                if meta.path.is_ident("iden") {
119                    let litstr: LitStr = meta.value()?.parse()?;
120                    new_iden = syn::parse_str::<TokenStream>(&litstr.value())?;
121                } else {
122                    // Reads the value expression to advance the parse stream.
123                    // Some parameters do not have any value,
124                    // so ignoring an error occurred here.
125                    let _: Option<Expr> = meta.value().and_then(|v| v.parse()).ok();
126                }
127                Ok(())
128            })
129        })?;
130
131    // Currently we only support enums and unit structs
132    match data {
133        syn::Data::Enum(DataEnum { variants, .. }) => {
134            if variants.is_empty() {
135                Ok(TokenStream::new())
136            } else {
137                Ok(impl_iden_for_enum(&ident, variants))
138            }
139        }
140        syn::Data::Struct(DataStruct {
141            fields: Fields::Unit,
142            ..
143        }) => Ok(impl_iden_for_unit_struct(
144            &ident,
145            new_iden.to_string().as_str(),
146        )),
147        _ => Ok(quote_spanned! {
148            ident.span() => compile_error!("you can only derive DeriveIden on unit struct or enum");
149        }),
150    }
151}
