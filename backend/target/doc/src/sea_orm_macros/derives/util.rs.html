5pub(crate) fn field_not_ignored(field: &Field) -> bool {
6    for attr in field.attrs.iter() {
7        if let Some(ident) = attr.path().get_ident() {
8            if ident != "sea_orm" {
9                continue;
10            }
11        } else {
12            continue;
13        }
14
15        if let Ok(list) = attr.parse_args_with(Punctuated::<Meta, Comma>::parse_terminated) {
16            for meta in list.iter() {
17                if let Meta::Path(path) = meta {
18                    if let Some(name) = path.get_ident() {
19                        if name == "ignore" {
20                            return false;
21                        }
22                    }
23                }
24            }
25        }
26    }
27    true
28}

44pub(crate) fn escape_rust_keyword<T>(string: T) -> String
45where
46    T: ToString,
47{
48    let string = string.to_string();
49    if RUST_KEYWORDS.iter().any(|s| s.eq(&string)) {
50        format!("r#{string}")
51    } else if RUST_SPECIAL_KEYWORDS.iter().any(|s| s.eq(&string)) {
52        format!("{string}_")
53    } else {
54        string
55    }
56}

58/// Turn a string to PascalCase while escaping all special characters in ASCII words.
59///
60/// (camel_case is used here to match naming of heck.)
61///
62/// In ActiveEnum, string_value will be PascalCased and made
63/// an identifier in {Enum}Variant.
64///
65/// However Rust only allows for XID_Start char followed by
66/// XID_Continue characters as identifiers; this causes a few
67/// problems:
68///
69/// - `string_value = ""` will cause a panic;
70/// - `string_value` containing only non-alphanumerics will become `""`
71///   and cause the above panic;
72/// - `string_values`:
73///      - `"A B"`
74///      - `"A  B"`
75///      - `"A_B"`
76///      - `"A_ B"`
77///
78/// All shares the same identifier of `"AB"`;
79///
80/// This function does the PascelCase conversion with a few special escapes:
81/// - Non-Unicode Standard Annex #31 compliant characters will converted to their hex notation;
82/// - `"_"` into `"0x5F"`;
83/// - `" "` into `"0x20"`;
84/// - Empty strings will become special keyword of `"__Empty"`
85///
86/// Note that this does NOT address:
87///
88/// - case-sensitivity. String value "ABC" and "abc" remains
89///   conflicted after .camel_case().
90///
91/// Example Conversions:
92///
93/// ```ignore
94/// assert_eq!(camel_case_with_escaped_non_uax31(""), "__Empty");
95/// assert_eq!(camel_case_with_escaped_non_uax31(" "), "_0x20");
96/// assert_eq!(camel_case_with_escaped_non_uax31("  "), "_0x200x20");
97/// assert_eq!(camel_case_with_escaped_non_uax31("_"), "_0x5F");
98/// assert_eq!(camel_case_with_escaped_non_uax31("foobar"), "Foobar");
99/// assert_eq!(camel_case_with_escaped_non_uax31("foo bar"), "Foo0x20bar");
100/// ```
101pub(crate) fn camel_case_with_escaped_non_uax31<T>(string: T) -> String
102where
103    T: ToString,
104{
105    let additional_chars_to_replace: [char; 2] = ['_', ' '];
106
107    let mut rebuilt = string
108        .to_string()
109        .chars()
110        .enumerate()
111        .map(|(pos, char_)| {
112            if !additional_chars_to_replace.contains(&char_)
113                && match pos {
114                    0 => unicode_ident::is_xid_start(char_),
115                    _ => unicode_ident::is_xid_continue(char_),
116                }
117            {
118                char_.to_string()
119            } else {
120                format!("{:#X}", char_ as u32)
121            }
122        })
123        .reduce(
124            // Join the "characters" (now strings)
125            // back together
126            |lhs, rhs| lhs + rhs.as_str(),
127        )
128        .map_or(
129            // if string_value is ""
130            // Make sure the default does NOT go through camel_case,
131            // as the __ will be removed! The underscores are
132            // what guarantees this being special case avoiding
133            // all potential conflicts.
134            String::from("__Empty"),
135            |s| s.to_upper_camel_case(),
136        );
137
138    if rebuilt
139        .chars()
140        .next()
141        .map(char::is_numeric)
142        .unwrap_or(false)
143    {
144        rebuilt = String::from("_") + &rebuilt;
145    }
146
147    rebuilt
148}

162pub(crate) trait GetMeta {
163    fn exists(&self, k: &str) -> bool;
164    fn get_as_kv(&self, k: &str) -> Option<String>;
165}
166
167impl GetMeta for Meta {
168    fn exists(&self, key: &str) -> bool {
169        let Meta::Path(path) = self else {
170            return false;
171        };
172        path.is_ident(key)
173    }
174
175    fn get_as_kv(&self, key: &str) -> Option<String> {
176        let Meta::NameValue(MetaNameValue {
177            path,
178            value: syn::Expr::Lit(exprlit),
179            ..
180        }) = self
181        else {
182            return None;
183        };
184
185        let syn::Lit::Str(litstr) = &exprlit.lit else {
186            return None;
187        };
188
189        if path.is_ident(key) {
190            Some(litstr.value())
191        } else {
192            None
193        }
194    }
195}

197#[cfg(test)]
198mod tests {
199    use super::*;
200
201    #[test]
202    fn test_non_uax31_escape() {
203        // Test empty string
204        assert_eq!(camel_case_with_escaped_non_uax31(""), "__Empty");
205
206        // Test additional_chars_to_replace (to_camel_case related characters)
207        assert_eq!(camel_case_with_escaped_non_uax31(" "), "_0x20");
208
209        // Test additional_chars_to_replace (multiples. ensure distinct from single)
210        assert_eq!(camel_case_with_escaped_non_uax31("  "), "_0x200x20");
211
212        // Test additional_chars_to_replace (udnerscores)
213        assert_eq!(camel_case_with_escaped_non_uax31("_"), "_0x5F");
214
215        // Test typical use case
216        assert_eq!(camel_case_with_escaped_non_uax31("foobar"), "Foobar");
217
218        // Test spaced words distinct from non-spaced
219        assert_eq!(camel_case_with_escaped_non_uax31("foo bar"), "Foo0x20bar");
220
221        // Test underscored words distinct from non-spaced and spaced
222        assert_eq!(camel_case_with_escaped_non_uax31("foo_bar"), "Foo0x5Fbar");
223
224        // Test leading numeric characters
225        assert_eq!(camel_case_with_escaped_non_uax31("1"), "_0x31");
226
227        // Test escaping also works on full string following lead numeric character
228        // This was previously a fail condition.
229        assert_eq!(
230            camel_case_with_escaped_non_uax31("1 2 3"),
231            "_0x310x2020x203"
232        );
233
234        assert_eq!(camel_case_with_escaped_non_uax31("씨오알엠"), "씨오알엠");
235
236        assert_eq!(camel_case_with_escaped_non_uax31("A_B"), "A0x5Fb");
237
238        assert_eq!(camel_case_with_escaped_non_uax31("AB"), "Ab");
239    }
